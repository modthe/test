How to Configure Google Sheets Integration

To connect your quiz app to Google Sheets, you need to create a Google Apps Script web app:

  1. Create a new Google Sheet
  2. Click on "Extensions" > "Apps Script"
  3. Replace the default code with the provided script
  4. Deploy as a web app with "Execute as me" and "Anyone" access
  5. Copy the web app URL and paste it in the field below

// Google Apps Script code for your Google Sheet

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var data = JSON.parse(e.postData.contents);

  if (data.type === "question") {
    var sheet = ss.getSheetByName("Questions");
    if (!sheet) {
      sheet = ss.insertSheet("Questions");
      sheet.appendRow(["Timestamp", "Question", "Answer1", "Answer2", "Answer3", "Answer4", "Correct"]);
    }
    
    var timestamp = new Date();
    sheet.appendRow([
      timestamp,
      data.question,
      data.answers[0],
      data.answers[1],
      data.answers[2],
      data.answers[3],
      data.correct
    ]);
    
    return ContentService.createTextOutput(
      JSON.stringify({ status: "success", message: "Question added" })
    ).setMimeType(ContentService.MimeType.JSON);
  }

  if (data.type === "settings") {
    var settingsSheet = ss.getSheetByName("Settings");
    if (!settingsSheet) {
      settingsSheet = ss.insertSheet("Settings");
      settingsSheet.appendRow(["Timestamp", "QuizDate", "QuizTime"]);
    }
    
    var timestamp = new Date();
    // Clear old settings (only keep latest)
    settingsSheet.clearContents();
    settingsSheet.appendRow(["Timestamp", "QuizDate", "QuizTime"]);
    settingsSheet.appendRow([timestamp, data.date, data.time]);

    return ContentService.createTextOutput(
      JSON.stringify({ status: "success", message: "Settings updated" })
    ).setMimeType(ContentService.MimeType.JSON);
  }
  
  if (data.type === "result") {
    var resultsSheet = ss.getSheetByName("Results");
    if (!resultsSheet) {
      resultsSheet = ss.insertSheet("Results");
      resultsSheet.appendRow(["Timestamp", "Player", "Question", "Answer", "Correct", "ResponseTime", "Status"]);
    }
    
    var timestamp = new Date();
    resultsSheet.appendRow([
      timestamp,
      data.player,
      data.question,
      data.answer,
      data.correct,
      data.responseTime,
      data.status
    ]);
    
    return ContentService.createTextOutput(
      JSON.stringify({ status: "success", message: "Result recorded" })
    ).setMimeType(ContentService.MimeType.JSON);
  }
}

function doGet(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  if (e.parameter.type === "questions") {
    var sheet = ss.getSheetByName("Questions");
    if (!sheet) {
      return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON);
    }
    
    var data = sheet.getDataRange().getValues();
    var result = [];
    for (var i = 1; i < data.length; i++) {
      result.push({
        timestamp: data[i][0],
        question: data[i][1],
        answers: [data[i][2], data[i][3], data[i][4], data[i][5]],
        correct: data[i][6]
      });
    }
    return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
  }

  if (e.parameter.type === "settings") {
    var sheet = ss.getSheetByName("Settings");
    if (!sheet || sheet.getLastRow() < 2) {
      return ContentService.createTextOutput(JSON.stringify({})).setMimeType(ContentService.MimeType.JSON);
    }
    
    var data = sheet.getDataRange().getValues();
    if (data.length > 1) {
      return ContentService.createTextOutput(
        JSON.stringify({
          timestamp: data[1][0],
          date: data[1][1],
          time: data[1][2]
        })
      ).setMimeType(ContentService.MimeType.JSON);
    }
  }
  
  // Default response
  return ContentService.createTextOutput(
    JSON.stringify({ error: "Invalid request type" })
  ).setMimeType(ContentService.MimeType.JSON);
}
